Wiederholung: Training und Test Split

Split generieren:

split <- initial_split(data = lin_model_data, prop = 0.9) 
data_train <- split %>% training()
data_test <- split %>% testing()

Approach 1: 

lin_model <- lm(attack ~., data = data_train)

pred <- lin_model %>%
  predict(data_test) %>%
  as_tibble()

pred <- pred$value %>% as.array()

yardstick::rmse_vec(data_test$attack,pred) 
[1] 0.2315013

Approach 2: 

lin_model_tidy <- linear_reg() %>%
  fit(attack ~., data = data_train)

lin_model_tidy %>%
  augment(data_test) %>%
  yardstick::rmse(attack,.pred)
